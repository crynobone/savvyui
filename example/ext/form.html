var form = new Js.ext.form;
			jQuery(function() {
				jQuery("#sample").submit(function() {
					Js.debug.log("start ext.form");
					
					var validate = form.validate(this);
					if(!!validate) {
						Js.util.Assert.isTrue(true, true, "Form has been validated, return following arguments: " + validate)
					} else {
						Js.util.Assert.isTrue(true, true, "Form not validated");
					}
					
					return false;
				});
			});
