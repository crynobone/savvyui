Js.base.create and Js.base.extend

			var person = Js.base.create({
				name: "",
				email: "",
				getName: function() {
					return this.name + " (" + this.email + ")";
				}
			});
			
			var zaki = new person;
			zaki.name = "Zaki";
			zaki.email = "crynobone@gmail.com";
			Js.util.Assert.isTrue(zaki.getName(), "Zaki (crynobone@gmail.com)", "zaki.getName() return " + zaki.getName());
			
			var student = person.extend({
				matric: "",
				__construct: function(name, email, matric) {
					this.name = name;
					this.email = email;
					this.matric = matric;
				},
				getName: function() {
					return this.name + ' / ' + this.email + ' / ' + this.matric;
				}
			});
			
			var nor = new student('Norhafiz', 'noriz@codenitive.com', '49070');
			Js.util.Assert.isTrue(nor.getName(), "Norhafiz / noriz@codenitive.com / 49070", "nor.getName() return " + nor.getName());
